ĐOẠN CODE 1 – DFS đường đi từ S đến T


#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1005;

int n, m, s, t;
bool a[MAXN][MAXN];
int visited[MAXN];

void enter()
{
    cin >> n >> m >> s >> t;
    int u, v;

    for (int i = 0; i < m; i++)
    {
        cin >> u >> v;
        a[u][v] = true;
        a[v][u] = true;
    }
}

void DFS(int u)
{
    for (int v = 1; v <= n; v++)
    {
        if (a[u][v] && visited[v] == 0)
        {
            visited[v] = u;
            DFS(v);
        }
    }
}

void xuly()
{
    if (visited[t] == 0)
        cout << -1;
    else
    {
        while (t != s)
        {
            cout << t << " ";
            t = visited[t];
        }
        cout << s;
    }
}

int main()
{
    enter();
    visited[s] = -1;
    DFS(s);
    xuly();
    return 0;
}

ĐOẠN CODE 2 – BFS đường đi từ S đến T


#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1001;
vector<int> a[MAXN];
bool visited[MAXN];

void bfs(int u)
{
    queue<int> q;
    q.push(u);
    visited[u] = true;

    while(!q.empty())
    {
        int v = q.front(); q.pop();
        for(int x : a[v])
        {
            if(!visited[x])
            {
                visited[x] = true;
                q.push(x);
            }
        }
    }
}

int main()
{
    int n, m, s, f;
    cin >> n >> m >> s >> f;

    for(int i = 0; i < m; i++)
    {
        int u, v;
        cin >> u >> v;
        a[u].push_back(v);
        a[v].push_back(u);
    }

    bfs(s);

    cout << (visited[f] ? "YES" : "NO");
    return 0;
}

ĐOẠN CODE 3 – DFS liệt kê thành phần liên thông chứa S


#include <bits/stdc++.h>
using namespace std;

const int MAX = 1005;
vector<int> adj[MAX];
bool visited[MAX];
vector<int> res;
int n, m, S;

void DFS(int u)
{
    visited[u] = true;
    res.push_back(u);

    for (int v : adj[u])
        if (!visited[v])
            DFS(v);
}

int main()
{
    cin >> n >> m >> S;

    for (int i = 0; i < m; i++)
    {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    for (int i = 1; i <= n; i++)
        sort(adj[i].begin(), adj[i].end());

    DFS(S);

    sort(res.begin(), res.end());
    for (int x : res)
        cout << x << " ";

    return 0;
}
