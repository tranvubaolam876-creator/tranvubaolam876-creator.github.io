#include 
using namespace std;

const int MOD = 1e9 + 7;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector grid(n + 1);
    for (int i = 1; i <= n; ++i)
        cin >> grid[i];

    vector> dp(n + 1, vector(n + 1, 0));

    if (grid[1][0] == '.') dp[1][1] = 1;  // chú ý: grid chỉ số 0-based trong string

    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= n; ++j) {
            if (grid[i][j - 1] == '*') {
                dp[i][j] = 0;
                continue;
            }
            if (i > 1) dp[i][j] = (dp[i][j] + dp[i - 1][j]) % MOD;
            if (j > 1) dp[i][j] = (dp[i][j] + dp[i][j - 1]) % MOD;
        }
    }

    cout << dp[n][n];
    return 0;
}

    int n;
    cin >> n;
    vector a(n);
    for (int i = 0; i < n; i++) cin >> a[i];
    
    long long prefix_sum = 0;
    long long min_prefix = 0; // tổng nhỏ nhất trước đó
    long long ans = LLONG_MIN;
    
    for (int i = 0; i < n; i++) {
        prefix_sum += a[i];
        ans = max(ans, prefix_sum - min_prefix);
        min_prefix = min(min_prefix, prefix_sum);
    }

    return 0;
